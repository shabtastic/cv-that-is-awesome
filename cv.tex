% Preserve the custom 'status' field so Biber passes it through to LaTeX.
% Without this, Biber silently drops unknown fields and \iffieldequalstr
% on 'status' always evaluates false (skipping all patent entries).
% \DeclareSourcemap fires during Biber's source-mapping phase and copies
% the raw 'status' value into 'addendum' (a known biblatex field).
% \iffieldequalstr then tests 'addendum' instead of 'status'.
\DeclareSourcemap{
  \maps[datatype=bibtex]{
    \map{
      \step[fieldsource=status, fieldtarget=addendum]
    }
    % Drop any entry tagged keywords={unlisted} during source-mapping so it
    % never reaches biblatex. Used for items kept in a bib file for the
    % corpus pipeline but intentionally excluded from every compiled CV
    % (e.g., PhD thesis in chapters.bib).
    \map{
      \step[fieldsource=keywords, match=\regexp{\bunlisted\b}, final]
      \step[entrynull]
    }
  }
}

% PUBLICATION ANNOTATION SYMBOLS
%
% \cvfnmark{equalcontrib} — superscript * (equal contributions / shared 1st)
% \cvfnmark{mentee}       — superscript † (mentee co-author)
%
% Used in usera fields and emitted automatically by the default name format
% for surnames listed in \cvmentees.
%
% The legend is printed once at the top of the Publications section via
% \cvpublegend, not repeated per subsection.
%------------------------------------------------------------------------------
\newcommand{\cvfnmark}[1]{%
  \IfStrEq{#1}{equalcontrib}{\textsuperscript{*}}{%
  \IfStrEq{#1}{mentee}{\textsuperscript{\textdagger}}{%
  \textsuperscript{?}}}%
}

% \cvpublegend: prints the symbol legend on its own line below the section heading.
% \bodyfontlight = Roboto-Thin which has no italic variant, so \textit silently
% falls back to upright. \bodyfont = Roboto-Regular which does have Roboto-Italic.
\newcommand{\cvpublegend}{%
  \par\vspace{-4pt}%
  {\bodyfont\fontsize{9pt}{1em}\selectfont\textit{* equal contributions, shared first authorship,\quad\textdagger\ mentee co-author}}%
  \par%
}

%==============================================================================
% BIBLIOGRAPHY NUMBERING TOGGLES & MACHINERY
%==============================================================================
%
% \toggletrue{showBibNumbers}    → show [1], [2], … labels
% \togglefalse{showBibNumbers}   → hide labels entirely
%
% \toggletrue{reverseBibNumbers} → N…1 (most recent = highest number)
% \togglefalse{reverseBibNumbers}→ 1…N (forward order)
%
% Numbering restarts at 1 (or N) for each bibliography section.
%
% Implementation: we use our own per-item counter entirely independent of
% biblatex's internal numbering. \AtEveryBibitem increments a running index;
% we write each section's total to the .aux so the next pass can compute
% reverse labels. No refsection environments needed.
%==============================================================================
\newtoggle{showBibNumbers}
\newtoggle{reverseBibNumbers}

\toggletrue{showBibNumbers}     % true = show numbers | false = hide numbers
\togglefalse{reverseBibNumbers} % true = reverse (N…1) | false = forward (1…N)

% Per-section counters
\newcounter{cvbibindex}   % increments with each printed entry (reset per section)
\newcounter{cvbibtotal}   % total entries in current section (loaded from aux)

% cvbibindex starts at 1. \AtEveryBibitem fires AFTER the label is rendered,
% so the label reads the current value (1, 2, 3…) and then the increment
% advances it for the next entry.
\AtEveryBibitem{\stepcounter{cvbibindex}}

% Override the printed label to use our own counter.
% labelnumber controls the number inside brackets in the entry body.
% We also override the list \item label via the labelnumberwidth format
% and the blx@lbx numeric label macro.
\DeclareFieldFormat{labelnumber}{%
  \iftoggle{showBibNumbers}{%
    \iftoggle{reverseBibNumbers}{%
      \number\numexpr\value{cvbibtotal}-\value{cvbibindex}+1\relax%
    }{%
      \number\value{cvbibindex}%
    }%
  }{%
    \hspace{-\labelsep}%
  }%
}

% Also patch the label used for \item[...] in the bibliography list env.
% In biblatex's numeric style this is \printtext[labelnumberwidth]{...}
% which calls \mkbibbrackets{\printfield{labelnumber}} — so overriding
% labelnumber above is sufficient.

% Redefine the bibliography list environment to control indentation.
% When numbers are shown: label in a compact gutter, text flush after.
% When numbers are hidden: no gutter, hanging indent of \bibhang.
\AtBeginDocument{%
  \iftoggle{showBibNumbers}{%
    \setlength{\bibhang}{0pt}%
    \defbibenvironment{bibliography}
      {\list{\mkbibbrackets{\printfield{labelnumber}}}
         {\setlength{\labelwidth}{2em}%
          \setlength{\labelsep}{0.5em}%
          \setlength{\leftmargin}{\labelwidth}%
          \addtolength{\leftmargin}{\labelsep}%
          \setlength{\itemindent}{0pt}%
          \setlength{\itemsep}{\bibitemsep}%
          \setlength{\parsep}{\bibparsep}}}
      {\endlist}
      {\item}%
  }{%
    \setlength{\biblabelsep}{0pt}%
    \defbibenvironment{bibliography}
      {\list{}
         {\setlength{\leftmargin}{\bibhang}%
          \setlength{\itemindent}{-\leftmargin}%
          \setlength{\itemsep}{\bibitemsep}%
          \setlength{\parsep}{\bibparsep}}}
      {\endlist}
      {\item}%
  }%
}

% \startbibsection{name} — call before each \printbibliography block.
%   Resets the index counter and loads the stored total for this section.
%   On the first compile the total is 0 (labels may be wrong); correct on
%   subsequent compiles once \finishbibsection has written the total to .aux.
\newcommand{\startbibsection}[1]{%
  \setcounter{cvbibindex}{1}%
  \ifcsdef{cvbibtotal@#1}{%
    \setcounter{cvbibtotal}{\csname cvbibtotal@#1\endcsname}%
  }{%
    \setcounter{cvbibtotal}{0}%
  }%
}

% \finishbibsection stores (cvbibindex - 1) as the total, since cvbibindex
% ends up one past the last entry after the final increment.
\newwrite\cvbibcountsfile
\AtBeginDocument{\immediate\openout\cvbibcountsfile=cvbibcounts.tex}
\AtEndDocument{\immediate\closeout\cvbibcountsfile}

\newcommand{\finishbibsection}[1]{%
  \immediate\write\cvbibcountsfile{%
    \string\expandafter\string\def
    \string\csname\space cvbibtotal@#1\string\endcsname
    {\number\numexpr\value{cvbibindex}-1\relax}%
  }%
}

% \cvtestmentee{<surname-macro>}: emit \cvfnmark{mentee} if the macro's
% expansion exactly matches one of the comma-separated items in \cvmentees.
% Iterates with \@for and uses \ifx (which compares macro replacement text)
% so the test is exact-string equality, not substring.
\makeatletter
\newcommand{\cvtestmentee}[1]{%
  \def\cv@menteefound{0}%
  \@for\cv@menteeitem:=\cvmentees\do{%
    \edef\cv@menteeitemexp{\cv@menteeitem}%
    \ifx\cv@menteeitemexp#1%
      \def\cv@menteefound{1}%
    \fi
  }%
  \ifnum\cv@menteefound=1 \cvfnmark{mentee}\fi
}
\makeatother

% Redefine the 'default' name format (used by all numeric-style entries via
% the sortname alias) to bold the CV owner's name.
\AtBeginDocument{%
  % Redefine the 'default' name format to bold the CV owner's name.
  % Must be in \AtBeginDocument so it fires after biblatex and numeric.bbx
  % have finished setting their own defaults.
  %
  % Comparison technique: \ifstrequal does NOT expand its arguments (it
  % compares literal token streams, not string values). \namepartfamily and
  % \cvownfamilyname are macros, so we \edef both into plain temporaries
  % first, then use \ifx — which compares macro replacement texts — to
  % get a reliable string equality test.
  \DeclareNameFormat{default}{%
    \edef\cvboldtmpA{\namepartfamily}%
    \edef\cvboldtmpB{\cvownfamilyname}%
    \ifx\cvboldtmpA\cvboldtmpB
      % Print delimiter OUTSIDE the bold wrapper, then bold just the name.
      % name:family-given includes name:delim internally, so we call
      % name:delim first here, then replicate the name output inside cvbibname.
      \usebibmacro{name:delim}{\namepartfamily}%
      \cvbibname{%
        \mkbibnamefamily{\namepartfamily}\isdot
        \ifdefvoid{\namepartgiveni}{}{\revsdnamepunct\bibnamedelimd\mkbibnamegiven{\namepartgiveni}\isdot}%
      }%
    \else
      \usebibmacro{name:family-given}{\namepartfamily}{\namepartgiveni}%
                                     {\namepartprefix}{\namepartsuffix}%
      % Mentee marker: append \cvfnmark{mentee} if surname is in \cvmentees.
      \cvtestmentee{\cvboldtmpA}%
    \fi
    \usebibmacro{name:andothers}%
  }%
}

%------------------------------------------------------------------------------
% \cvsubsection — patched for consistent leading spacing
%
% Awesome-CV's default \cvsubsection (cls) does:
%   \vspace{2.5mm} + \vspace{-3mm} + text  → net -0.5mm
%
% This is fine when subsection follows another biblatex bibliography (tight
% tuck inside Publications), but produces zero (or even negative) leading
% space when subsection follows \end{cventries} (e.g. Mentorship subsections)
% or appears as the first thing after \cvsection (horizontal-mode bug —
% vspaces get ignored and the heading flows inline).
%
% We patch in two parts:
%   1. \par\noindent before the vspaces — fixes the horizontal-mode bug so
%      vspaces actually take effect; \noindent suppresses extra \parskip.
%   2. Conditional leading gap — \cvsubsectionLeadGap pt of breathing room
%      when subsection follows a \begin{cventries} block. AtEndEnvironment
%      sets \cvprevwasentries; cvsubsection consumes and clears it.
%------------------------------------------------------------------------------
% \cvsubsectionLeadGap   — always added before any \cvsubsection
% \cvsubsectionAfterEntries — extra gap when subsection follows \end{cventries}
\newcommand{\cvsubsectionLeadGap}{4pt}
\newcommand{\cvsubsectionAfterEntries}{6pt}
\newbool{cvprevwasentries}
\boolfalse{cvprevwasentries}
\AtEndEnvironment{cventries}{\booltrue{cvprevwasentries}}

\patchcmd{\cvsubsection}%
  {\vspace{\acvSectionContentTopSkip}}%
  {\par\noindent\vspace{\cvsubsectionLeadGap}\ifbool{cvprevwasentries}{\vspace{\cvsubsectionAfterEntries}\boolfalse{cvprevwasentries}}{}\vspace{\acvSectionContentTopSkip}}%
  {}{}

\defbibheading{cvsubsection}[\bibname]{%
  \cvsubsection{#1}%
}

                           heading=cvsubsection]%

                           heading=cvsubsection]%

                           title={Filed},
                           heading=cvsubsection]%

    }%
  }%

% Enforce consistent vertical spacing between \cventry blocks.
%
% Root cause of the indent bug: \apptocmd{\cventry}{\vspace{N}} appends a
% \vspace after the tabular. Inside \begin{center}, this leaves a vertical-mode
% artefact that makes the NEXT \cventry's tabular shift right. Using \pretocmd
% instead — and skipping the gap for the first entry in each block — avoids
% any post-tabular state and keeps all entries flush-left.
%
% Gap value: 0pt — the natural tabular bottom-padding (~15pt) provides
% sufficient visual separation between entries. Any positive value here adds
% on top of that and makes gaps too large in dense sections like Teaching/Service.
\newbool{firstcventry}
\newbool{prevhaddesc}
\booltrue{firstcventry}
\boolfalse{prevhaddesc}
\AtBeginEnvironment{cventries}{%
  \booltrue{firstcventry}%
  \boolfalse{prevhaddesc}%
  \setlength{\parskip}{0pt}%
  \setlength{\parindent}{0pt}%
}
% Before each \cventry: add the right vertical gap based on context.
%   - First entry in a cventries block: no leading gap.
%   - Previous entry had bullets (cvitems): \cvgapAfterBullets — clears the
%     bullet content which extends below the tabular's measured height.
%   - Plain entry → plain entry: \cvgapPlain — matches cls default.
\newcommand{\cvgapPlain}{4pt}
\newcommand{\cvgapAfterBullets}{6pt}
\pretocmd{\cventry}{%
  \ifbool{firstcventry}{%
    \boolfalse{firstcventry}%
  }{%
    \ifbool{prevhaddesc}{%
      \vspace{\cvgapAfterBullets}%
    }{%
      \vspace{\cvgapPlain}%
    }%
  }%
  \boolfalse{prevhaddesc}%
  \noindent%
}{}{}
% After each \cventry: record whether arg #5 was non-empty (had description).
% We can't inspect #5 directly in apptocmd, so we patch cvitems end instead.
\AtEndEnvironment{cvitems}{\booltrue{prevhaddesc}}

% cvitems: bullet block inside a \cventry's 5th argument (description cell).
%
% Opening \vspace pulls bullets up close to the entry's position line.
% Closing \vspace inflates the cell so its measured bottom matches where
% the bullets actually end — without it, bullet content overflows the
% tabular row and the next entry's header overlaps it. \cvgapAfterBullets
% in pretocmd handles inter-entry breathing room on top of that.
\renewenvironment{cvitems}{%
  \vspace{-3.0mm}%
  \begin{justify}%
  \begin{itemize}[leftmargin=2ex, nosep, noitemsep]%
    \setlength{\parskip}{0pt}%
    \renewcommand{\labelitemi}{\bullet}%
}{%
  \end{itemize}%
  \end{justify}%
  \vspace{2mm}%
}

%-- Publications --------------------------------------------------------------
% Top-level heading: "Publications" or "Selected Publications"
% Subsections: Journals · Preprints · Conference Proceedings ·
%              Conference Presentations · Book Chapters
%
% Toggle \toggletrue{selectedOnlyPublications} to switch the section title.
% Individual subsection selected-only filters remain independent.

\newbool{showPublications}
\boolfalse{showPublications}
\iftoggle{showJournals}{\booltrue{showPublications}}{}
\iftoggle{showPreprints}{\booltrue{showPublications}}{}
\iftoggle{showConferenceProceedings}{\booltrue{showPublications}}{}
\iftoggle{showConferencePresentations}{\booltrue{showPublications}}{}
\iftoggle{showBookChapters}{\booltrue{showPublications}}{}
\iftoggle{showSciComm}{\booltrue{showPublications}}{}

\ifbool{showPublications}{%
  \iftoggle{selectedOnlyPublications}{%
    \cvsection{Selected Publications}%
  }{%
    \cvsection{Publications}%
  }%
  \cvpublegend%

  \iftoggle{showJournals}{%
    \startbibsection{journals}%
    \iftoggle{selectedOnlyJournals}{%
      \leavevmode\printbibliography[type=article, keyword=selected,
                         check=notcommentary,
                         title={Peer-Reviewed Journal Publications (Selected)},
                         heading=cvsubsection]%
    }{%
      \leavevmode\printbibliography[type=article,
                         check=notcommentary,
                         title={Peer-Reviewed Journal Publications},
                         heading=cvsubsection]%
    }%
    \finishbibsection{journals}%
  }{}

  \iftoggle{showPreprints}{%
    \startbibsection{preprints}%
    \iftoggle{selectedOnlyPreprints}{%
      \leavevmode\printbibliography[type=unpublished, keyword=selected,
                         check=notpresentation,
                         title={Preprints \& Working Papers (Selected)},
                         heading=cvsubsection]%
    }{%
      \leavevmode\printbibliography[type=unpublished,
                         check=notpresentation,
                         title={Preprints \& Working Papers},
                         heading=cvsubsection]%
    }%
    \finishbibsection{preprints}%
  }{}

  \iftoggle{showConferenceProceedings}{%
    \startbibsection{conference}%
    \iftoggle{selectedOnlyConferenceProceedings}{%
      \leavevmode\printbibliography[type=inproceedings, keyword=selected,
                         title={Peer-Reviewed Conference Proceedings (Selected)},
                         heading=cvsubsection]%
    }{%
      \leavevmode\printbibliography[type=inproceedings,
                         title={Peer-Reviewed Conference Proceedings},
                         heading=cvsubsection]%
    }%
    \finishbibsection{conference}%
  }{}

  \iftoggle{showConferencePresentations}{%
    \startbibsection{presentations}%
    \iftoggle{selectedOnlyPresentations}{%
      % Use a bibcheck for AND logic (keyword=presentation AND keyword=selected),
      % since multiple keyword= filters in \printbibliography are OR not AND.
      \leavevmode\printbibliography[check=presentationselected,
                         title={Selected Conference Presentations},
                         heading=cvsubsection]%
    }{%
      \leavevmode\printbibliography[keyword=presentation,
                         title={Conference Presentations},
                         heading=cvsubsection]%
    }%
    \finishbibsection{presentations}%
  }{}

  \iftoggle{showBookChapters}{%
    \startbibsection{bookchapters}%
    \leavevmode\printbibliography[type=incollection,
                       title={Reviews, Book Chapters \& Commentaries},
                       heading=cvsubsection]%
    % Commentary articles (e.g. AI Magazine, newsletters) live in
    % refs/chapters.bib tagged keywords={commentary}. Print them here under
    % the same heading, continuing the numbering from the incollection block.
    \leavevmode\printbibliography[type=article, keyword=commentary,
                       heading=none]%
    \finishbibsection{bookchapters}%
  }{}

  % Science Communication — always at end of Publications section
  \iftoggle{showSciComm}{%
    \startbibsection{scicomm}%
    \leavevmode\printbibliography[type=misc, keyword=scicomm,
                       title={Science Communication},
                       heading=cvsubsection]%
    \finishbibsection{scicomm}%
  }{}%
}{}
