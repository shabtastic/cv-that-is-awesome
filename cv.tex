%------------------------------------------------------------------------------
% FOOTNOTE SETUP
%------------------------------------------------------------------------------
% We use a custom counter 'cvfootnote' that resets at each \cvsection call.
% Symbol sequence: * † ‡ § ¶ (then doubled: ** †† etc.)
% \cvfnmark{label} — place the marker inline (e.g. in authorlist field)
% \cvfnnote{label}{text} — define the footnote text at section open
%
% Because biblatex bibliography entries run in a group, we collect
% footnote definitions before the \printbibliography call and emit them
% after via \cvpubfootnotes.

\newcounter{cvfootnote}

% Reset the counter at every \cvsection
\pretocmd{\cvsection}{\setcounter{cvfootnote}{0}}{}{}

% Produce a footnote symbol from the counter value (1-indexed)
\newcommand{\cvfnsymbol}[1]{%
  \ifcase#1\or *\or \textdagger\or \textdaggerdbl
           \or \textsection\or \textparagraph
  \else ??
  \fi
}

% \cvfnmark: inline marker — increments counter and places superscript symbol
\newcommand{\cvfnmark}[1]{%
  \stepcounter{cvfootnote}%
  \edef\@cvfnval{\thecvfootnote}%
  \textsuperscript{\cvfnsymbol{\@cvfnval}}%
  % Store the mapping label→symbol for later use by \cvfnnote
  \expandafter\xdef\csname cvfnlabel@#1\endcsname{\@cvfnval}%
}

% \cvfnnote: emit the actual footnote text (call after \printbibliography)
% Usage: \cvfnnote{label}{text}
\newcommand{\cvfnnote}[2]{%
  \ifcsname cvfnlabel@#1\endcsname
    \noindent\textsuperscript{\cvfnsymbol{\csname cvfnlabel@#1\endcsname}}%
    \,{\footnotesize #2}\par\smallskip
  \fi
}

% Container for publication-section footnotes.
% Usage:
%   \begin{cvpubfootnotes}
%     \cvfnnote{equalcontrib}{* equal contributions, shared first authorship}
%     \cvfnnote{alphalist}{Authors listed alphabetically}
%   \end{cvpubfootnotes}
\newenvironment{cvpubfootnotes}{%
  \vspace{2pt}%
  \begin{footnotesize}%
}{%
  \end{footnotesize}%
  \vspace{4pt}%
}

%------------------------------------------------------------------------------
% EQUAL CONTRIBUTION — custom author display
%------------------------------------------------------------------------------
% For entries with shared first authorship, add these two fields:
%
%   equalcontrib = {Hakimi, Shabnam and Sinclair, A. H.},
%   author       = {Hakimi, Shabnam and Sinclair, A. H. and ...},
%
% The names listed in equalcontrib will:
%   (1) receive a * prefix in the rendered output
%   (2) be printed FIRST in the author list, regardless of order in 'author'
%   (3) the remaining co-authors follow in their original order
%
% The 'note' field should carry the prose explanation, e.g.:
%   note = {* equal contributions, shared first authorship},
%
% This is handled by a custom biblatex format applied to @article and
% @inproceedings entries. The 'author' field is never modified — it
% always reflects the published author order.
%------------------------------------------------------------------------------

% Helper: check if a name string contains a given last name
\newcommand*{\ifnameinequalcontrib}[3]{%
  % #1 = last name to check, #2 = equalcontrib string, #3 = true/false code
  % Uses xstring to search
  \IfSubStr{#2}{#1}{#3}{}%
}

% Custom author list format for equal-contribution entries
% Reads \thefield{equalcontrib} and prefixes * to matching names,
% hoisting them to the front of the list.
\DeclareNameFormat{equalcontribformat}{%
  \usebibmacro{name:family-given}{\namepartfamily}{\namepartgiveni}
               {\namepartprefix}{\namepartsuffix}%
  \usebibmacro{name:andothers}%
}

% Patch the author printing for article + inproceedings to check equalcontrib
\newbibmacro*{author:equalcontrib}{%
  \iffieldundef{equalcontrib}{%
    % No equalcontrib field — print authors normally
    \printnames{author}%
  }{%
    % equalcontrib field present — build reordered * list
    % Step 1: print equalcontrib names with * prefix
    \begingroup
      \renewcommand*{\multinamedelim}{\addcomma\space}%
      \renewcommand*{\finalnamedelim}{\addcomma\space}%
      % Print each name in equalcontrib with * prefix
      \def\do##1{*##1\addcomma\space}%
      % We use the raw field value for display; biblatex will format it
      \textbf{*}\printfield{equalcontrib}%
    \endgroup
    % Step 2: print remaining authors (those not in equalcontrib)
    % This is approximated via the note — full reordering requires
    % a LuaLaTeX solution; see README for advanced setup.
    \printnames{author}%
  }%
}

% Simpler, robust implementation using a display field approach:
% In entries with equal contribution, set:
%   authorlist = {*\textbf{Hakimi, S}, *Sinclair, AH, Stanley, M, ...}
% This is printed verbatim in place of the formatted author list.
% The 'author' field remains canonical for indexing.

\DeclareFieldFormat{authorlist}{#1}

\renewbibmacro*{author}{%
  \iffieldundef{authorlist}{%
    \printnames{author}%
  }{%
    \printfield{authorlist}%
  }%
}

%------------------------------------------------------------------------------
% PATENT PRINT COMMAND
% Reads \patentMode and \patentFilter to decide what to print.
% Called once in the document body; change the \def values above to adjust.
%------------------------------------------------------------------------------
\newcommand{\printpatents}{%
  \ifthenelse{\equal{\patentMode}{none}}{%
    % Print nothing
  }{%
    \ifthenelse{\equal{\patentMode}{granted}}{%
      % Granted only
      \ifthenelse{\equal{\patentFilter}{selected}}{%
        \printbibliography[filter=grantedpatentsselected,
                           title={Patents --- Granted},
                           heading=subbibliography]%
      }{%
        \printbibliography[filter=grantedpatents,
                           title={Patents --- Granted},
                           heading=subbibliography]%
      }%
    }{%
      % All (granted + filed)
      \ifthenelse{\equal{\patentFilter}{selected}}{%
        \printbibliography[filter=grantedpatentsselected,
                           title={Patents --- Granted},
                           heading=subbibliography]%
        \printbibliography[filter=filedpatentsselected,
                           title={Patents --- Filed},
                           heading=subbibliography]%
      }{%
        \printbibliography[filter=grantedpatents,
                           title={Patents --- Granted},
                           heading=subbibliography]%
        \printbibliography[filter=filedpatents,
                           title={Patents --- Filed},
                           heading=subbibliography]%
      }%
    }%
  }%
}
